Complete Example — Grade Calculator

Grade Calculator — uses every major feature

-- Grade Calculator

store "Alice" into student
store collection 88 and 92 and 75 and 98 and 84 into scores

-- function: compute average
build average using nums
  store result of sum with nums into total
  store result of length with nums into count
  return total / count
done

-- function: assign letter grade
build grade using avg
  when avg is at least 90 then
    return "A"
  elif avg is at least 80 then
    return "B"
  elif avg is at least 70 then
    return "C"
  otherwise
    return "F"
  done
done

-- calculate and display
store result of average with scores into avg
store result of grade with avg into letter
output "Student: " joined with student
output "Average: " joined with avg
output "Grade:   " joined with letter

-- count scores above 85
store 0 into high
go through every s in scores
  when s is bigger than 85 then
    increment high by 1
  done
done
output "Scores above 85: " joined with high
